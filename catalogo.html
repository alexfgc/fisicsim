// 1. Extraemos todas las asignaturas únicas // que aparecen en el array, en orden de // aparición. const asignaturas = [...new Set(SIMULACIONES.map(s => s.asignatura))]; const contenedor = document.getElementById("catalogo-contenido"); // 2. Por cada asignatura, creamos una sección // con su título y un grid de tarjetas. asignaturas.forEach(asignatura => { // Filtramos solo las sims de esta asignatura const simsDeEstaAsignatura = SIMULACIONES.filter( s => s.asignatura === asignatura ); // Creamos el HTML de la sección const seccion = document.createElement("section"); seccion.className = "catalogo-seccion"; seccion.innerHTML = `

${asignatura}

${simsDeEstaAsignatura.map(sim => `
⟳

${sim.nombre}

${sim.descripcion}

${sim.tags.map(tag => `${tag}`).join("")}
`).join("")}
`; contenedor.appendChild(seccion); });
